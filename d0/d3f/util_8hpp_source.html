229 template<typename T>
230 T convertToSeconds(const std::string &unit, T value) {
231  T seconds;
232  if (unit == "min") {
233  seconds = value * 60;
234  } else if (unit == "h") {
235  std::string new_unit = "min";
236  seconds = convertToSeconds(new_unit, value * 60);
237  } else if (unit == "s" || unit == "sec") {
238  seconds = value;
239  } else if (isScalable(unit, "s")) {
240  seconds = value * getSIScaling(unit, "s");
241  } else {
242  std::cerr << "[nix::util::convertToSeconds] Warning: given unit is not supported!" << std::endl;
243  seconds = value;
244  }
245  return seconds;
246 }

256 template<typename T>
257 T convertToKelvin(const std::string &unit, T value) {
258  T temperature;
259  if (unit == "°C" || unit == "C") {
260  temperature = value + 273.15;
261  } else if (unit == "°F" || unit == "F") {
262  double temp = (value - 32) * 5.0/9 + 273.15;
263  temperature = std::is_integral<T>::value ? std::round(temp) : temp;
264  } else if (unit == "°K" || unit == "K") {
265  temperature = value;
266  } else if (isScalable(unit, "K")) {
267  temperature = value * getSIScaling(unit, "K");
268  } else {
269  std::cerr << "[nix::util::convertToKelvin] Warning: given unit is not supported" << std::endl;
270  temperature = value;
271  }
272  return temperature;
273 }
