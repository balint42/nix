454 template<typename T>
455 void DataArray::getData(T &value) const
456 {
457  Hydra<T> hydra(value);
458 
459  NDSize extent = backend()->dataExtent();
460  hydra.resize(extent);
461 
462  DataType dtype = hydra.element_data_type();
463  NDSize shape = hydra.shape();
464 
465  backend()->read(dtype, hydra.data(), shape, {});
466 }

468 template<typename T>
469 void DataArray::setData(const T &value)
470 {
471  const Hydra<const T> hydra(value);
472 
473  DataType dtype = hydra.element_data_type();
474  NDSize shape = hydra.shape();
475 
476  if(!backend()->hasData()) {
477  backend()->createData(dtype, shape);
478  }
479  backend()->dataExtent(shape);
480  backend()->write(dtype, hydra.data(), shape, {});
481 }
